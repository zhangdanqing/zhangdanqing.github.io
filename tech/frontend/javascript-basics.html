类 ​

javascript
class Person {
    constructor(name, age) {
        this.name = name;
        this.age = age;
    }

    sayHello() {
        console.log(`Hello, I'm ${this.name}`);
    }
}

javascript
async function getData() {
    try {
        const result = await fetch('api/data');
        return await result.json();
    } catch (error) {
        console.error(error);
    }
}
